\begin{frame}{Motivation, question, headline}
\textbf{Setting.} Europe moved electricity trading from 60-min to 15-min. Spain: two reforms in 2025.

\bigskip
\textbf{Question.} Does finer granularity reshape \emph{strategic} bidding? Who, when, which channel, market power up or down?

\bigskip
\textbf{Headline.} Granularity is a \textcolor{blue!50!black}{\textbf{strategic-latitude reveal}}:
\begin{itemize}\setlength{\itemsep}{2pt}
  \item Dispatchables widen ladders in critical hours.
  \item Granular-side prices fall; other market follows by anticipation.
  \item Wedge dispersion jumps and does not collapse.
\end{itemize}

\bigskip
$\Rightarrow$ Market power \textbf{compressed}, not amplified. Post-blackout cost rise is a separate story.
\end{frame}

\footnotesize
\begin{itemize}\setlength{\itemsep}{2pt}
  \item Three reforms in seven months: ISP15, ID15, DA15.
  \item April 2025 blackout $\to$ continuous \emph{reforzada} regime.
  \item ID15 windows: pre-blackout. DA15 windows: hold reforzada constant by design.
\end{itemize}

% ---------------------------------------------------------------------------
\begin{frame}{Model sketch}
\textbf{Setup.} Sequential Cournot, extends Ito \& Reguant (2016) with a granularity selector at each stage. Three agents: strategic bidder, operational aggregator, capacity-bound arbitrageur.

\bigskip
\textbf{Reform path:} $(1,1) \to (1,Q) \to (Q,Q)$.

\bigskip
\textbf{Key predictions to test:}
\begin{itemize}\setlength{\itemsep}{3pt}
  \item Bid ladders widen on the granular side.
  \item Aggregator shifts quantity into the granular market.
  \item Granular-side price drops; other market drops by anticipation.
  \item Cournot scale-up at $(Q,Q)$.
  \item Wedge SD does \emph{not} collapse at $(Q,Q)$ --- limited per-product arbitrage.
  \item Production-unit arbitrageurs exist and bind.
\end{itemize}
\end{frame}

\begin{column}{0.50\textwidth}
\footnotesize
\textbf{In-band set} (per curve $u,d,p$):
\[
\mathcal{B} \;=\; \{k:\,|p_k - \text{MCP}_{d,p}| \le h\}, \quad
w_k \;=\; \frac{\Delta q_k}{\sum_{j\in\mathcal{B}} \Delta q_j}.
\]

\textbf{In-band weighted mean:}
\[
\bar p \;=\; \sum_{k\in\mathcal{B}} w_k\,p_k.
\]

\textbf{Bid-shape outcomes:}
\[
\sigma_p \;=\; \sqrt{\sum_{k\in\mathcal{B}} w_k\,(p_k - \bar p)^2},
\]
\[
N_{\text{eff}} \;=\; \Big(\sum_{k\in\mathcal{B}} w_k^2\Big)^{-1}.
\]

\bigskip
\textbf{Bandwidth:} $h = p_{90} - p_{50}$ of MCP per (window, market).
\end{column}

\begin{frame}{Identification}
\textbf{Data.} OMIE bid-level (2022--2026), ENTSO-E weather, ESIOS ajuste costs.

\bigskip
\textbf{Critical / flat partition} from 15-min residual demand $\sigma_{\text{within}}$:
\begin{itemize}\setlength{\itemsep}{2pt}
  \item Critical $=$ morning $+$ evening ramp.
  \item Flat $=$ overnight $\Rightarrow$ within-day low-exposure control.
\end{itemize}

\bigskip
\textbf{Three specifications:}
\begin{itemize}\setlength{\itemsep}{2pt}
  \item Spec A --- BSTS on prices, cleared MWh, wedge SD.
  \item Spec B --- per-hour-class BSTS on in-band offered energy.
  \item Spec C --- per-curve DiD on bid shape, critical $-$ flat differential.
\end{itemize}

\bigskip
\textbf{Same-calendar placebos} (2024, plus 2026 for ID15) net out recurring calendar shifts. DA15 pre-window starts at the blackout $\Rightarrow$ holds \emph{reforzada} constant.
\end{frame}

\end{itemize}
