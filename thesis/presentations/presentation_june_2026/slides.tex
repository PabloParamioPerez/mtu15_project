\begin{frame}{Motivation, question, headline}
\textbf{Setting.} Europe moved electricity trading from 60-min to 15-min. Spain: two reforms in 2025.

\bigskip
\textbf{Question.} Does finer granularity reshape \emph{strategic} bidding? Who, when, which channel, market power up or down?

\bigskip
\textbf{Headline.} Granularity is a \textcolor{blue!50!black}{\textbf{strategic-latitude reveal}}. Four empirically identified channels:
\begin{itemize}\setlength{\itemsep}{3pt}
  \item Dispatchables widen ladders in critical hours (bid-shape DiD).
  \item Within-hour DA$-$IDA wedge SD jumps under ID15 and does not collapse at DA15 (limited per-product arbitrage).
  \item IDA sell-share rises sharply at ID15 for nuclear, wind, pump-storage (production-unit arbitrageurs reposition).
  \item Auction-stage granularity shrinks REE's ajuste cost; the post-blackout cost rise is a separate reforzada response.
\end{itemize}

\bigskip
$\Rightarrow$ Market power \textbf{compressed} on the identified channels; not amplified.
\end{frame}

\begin{frame}{Model --- aggregator allocates between forward and spot}
\footnotesize
\textbf{Problem.} Per clock-hour, split $\bar Y_h$ between forward MWh and per-subperiod spot MWh. Quadratic imbalance penalty $\gamma \mathcal{I}_t^2$ per delivery product on any deviation from delivered $y_t$:
\[
\min_{\{Y^F, Y^S_t\}} \;\; p_1 Y^F + \sum_t p_{2t} Y^S_t + \gamma \sum_t E\big[(Y^F/Q + Y^S_t - y_t)^2\big] \quad \text{s.t.} \quad Y^F + \textstyle\sum_t Y^S_t = \bar Y_h.
\]

\textbf{Result (Appendix~\ref*{app:aggregator} of thesis).}
\[
\boxed{\;\dfrac{\partial \Delta^*}{\partial \sigma_\eta^2} \;>\; 0\,, \qquad \Delta^* \equiv \text{MWh shifted forward}\to\text{spot at }(1,Q).\;}
\]

\bigskip
\textbf{Intuition.} A per-quarter spot lets the aggregator submit a schedule that tracks its quarter-level forecast; the hourly forward locks one schedule per clock-hour. The hedging gain is increasing in forecast-error variance $\sigma_\eta^2$ --- the more uncertain you are, the more you want the finer scheduling tool.
\end{frame}

% ---------------------------------------------------------------------------
\begin{frame}{Model --- strategic bidder's two-tranche ladder}
\footnotesize
\textbf{Problem.} For each delivery product the bidder commits a ladder \emph{before} $\mu_t$ realises. Reduce-form to two tranches $(p_L, p_H)$ with $p_L < p_H$ and quantities $(q_L, q_H)$:
\[
\max_{p_L, p_H, q_L, q_H} \;\; E_{\mu_t}\big[(p_L - c)\, q_L\,\mathbf{1}\{p_L \le \text{MCP}_t\} + (p_H - c)\, q_H\,\mathbf{1}\{p_H \le \text{MCP}_t\}\big]
\]
where $\text{MCP}_t$ depends on $\mu_t$ through the per-product residual demand $A_t - b_{2t}\,p$.

\medskip
\textbf{Result.} Pre-realisation optimal spread (Appendix~\ref*{app:tranche} of thesis):
\[
\boxed{\;\Delta p^*_{2t} \;=\; p_H^* - p_L^* \;=\; \dfrac{\Delta_\mu}{2\, b_{2t}}.\;}
\]

\bigskip
\textbf{Intuition.} A wider spread captures more rent in tight states (high $\mu_t$) at the cost of foregone acceptance in slack states (low $\mu_t$). The optimum scales \emph{with the shock range} $\Delta_\mu$ and inversely with how steep per-product residual demand is. Ramp hours $\Rightarrow$ larger $\Delta_\mu$ $\Rightarrow$ wider ladders.
\end{frame}

% ---------------------------------------------------------------------------
\begin{frame}{Model --- arbitrage and the wedge SD at $(Q,Q)$}
\footnotesize
\textbf{Arbitrageur problem.} Choose position $s$ from forward to spot to maximise $\pi^A(s) = s\,(p_1 - p_2)$, possibly subject to capacity $|s| \le K$. Four regimes (Ito \& Reguant):

\smallskip
\begin{tabular}{l l}
(1) No arbitrage     & $s=0$, scarcity wedge $p_1>p_2$ \\
(2) Full arbitrage   & $p_1 = p_2$ \\
(3) Limited arbitrage& $s = K$ binds, wedge $\propto 1/K$ \\
(4) Strategic arb.   & $p_1 - p_2 = (p_1 - c)/4$ \\
\end{tabular}

\medskip
At $(Q,Q)$, capacity is split across $Q$ delivery products: per-product budget $K/Q$ binds tightly. \textbf{Result:}

\smallskip
\centering
\begin{tabular}{l c}
\toprule
Wedge SD, strategic (unlimited) & $\sigma_\mu / (8\, b_1)$ \\
Wedge SD, limited per-product   & $\boxed{\;\sigma_\mu / (4\, b_1)\;}$ \\
\bottomrule
\end{tabular}

\bigskip
\raggedright
\textbf{Intuition.} Naively, once both markets are per-quarter, arbitrage should equalise prices product-by-product. It doesn't, because real arbitrageurs have finite balance sheets and now have to allocate one budget across $Q$ separate products. Per-product capacity binds tightly $\Rightarrow$ residual wedge SD stays at \emph{twice} the unlimited-strategic benchmark.
\end{frame}

\begin{itemize}\setlength{\itemsep}{1pt}

\begin{column}{0.50\textwidth}
\footnotesize
\textbf{In-band set} (per curve $u,d,p$):
\[
\mathcal{B} \;=\; \{k:\,|p_k - \text{MCP}_{d,p}| \le \delta\}, \quad
w_k \;=\; \frac{\Delta q_k}{\sum_{j\in\mathcal{B}} \Delta q_j}.
\]

\textbf{In-band weighted mean:}
\[
\bar p \;=\; \sum_{k\in\mathcal{B}} w_k\,p_k.
\]

\textbf{Bid-shape outcomes (per curve):}
\[
\sigma_p \;=\; \sqrt{\sum_{k\in\mathcal{B}} w_k\,(p_k - \bar p)^2}, \quad
N_{\text{eff}} \;=\; \Big(\sum_{k\in\mathcal{B}} w_k^2\Big)^{-1}.
\]

\textbf{IDA in-band sell-share} (per tech $t$, per day $d$):
\[
\text{SS}_{t,d} \;=\; \frac{\sum_q q^{\text{sell,in-band}}_{t,d,q}}{\sum_q\big(q^{\text{sell,in-band}}_{t,d,q} + q^{\text{buy,in-band}}_{t,d,q}\big)}.
\]

\textbf{Bandwidth:} $\delta = p_{90} - p_{50}$ of MCP per (window, market).

\begin{frame}{Identification --- three specifications}
\footnotesize
\textbf{Spec~A: BSTS counterfactual on level outcomes} (clearing prices, per-tech cleared MWh, wedge SD, IDA sell-share).
\begin{itemize}\setlength{\itemsep}{1pt}
  \item Long pre-window (2022-01 onwards), covariates: $\text{wind}_t$, $\text{solar}_t$, $\text{gas}_t$, plus $\text{wind}_t \times \mathbf{1}\{\text{year}=y\}$ and $\text{solar}_t \times \mathbf{1}\{\text{year}=y\}$ for each $y$ in the window.
  \item Year-interaction is essential: Spanish wind+solar capacity grew $\sim\!6\times$ over 2018-2025; a flat renewable coefficient mis-fits the counterfactual and reports artificially confident effects.
  \item Same-calendar 2024 placebo nets recurring calendar shifts.
\end{itemize}

\medskip
\textbf{Spec~B: per-hour-class BSTS} on in-band offered energy per (tech, market, $\mathcal{X}$) cell, $\mathcal{X}\in\{$morning ramp, midday, evening ramp, flat$\}$. Same architecture as Spec~A.

\medskip
\textbf{Spec~C: per-curve DiD on bid shape.}
\[
y_{u,d,p} \;=\; \alpha_u + \beta\cdot\mathbf{1}\{d\ge T_r\} + \xi\cdot\mathbf{1}\{h(p)\in\mathcal{C}\} + \theta\cdot\mathbf{1}\{d\ge T_r,\,h(p)\in\mathcal{C}\} + \varepsilon
\]
on $y \in \{\sigma_p, N_{\text{eff}}\}$, SE clustered by date. Identification: the within-day critical/flat contrast acts as a control, absorbing calendar variation that Spec~A handles through the year-interactions.

\bigskip
\textbf{Reform windows.} ID15 post-window: 2025-03-19 $\to$ 2025-04-27 (pre-blackout, $40$d). DA15 post-window: 2025-10-01 $\to$ 2025-11-09 ($40$d). DA15 pre-window starts at the blackout $\Rightarrow$ \emph{reforzada} held constant by design.
\end{frame}

\begin{frame}{Conclusion --- granularity is a strategic-latitude reveal}
\footnotesize
\textbf{What we identify (four channels):}
\begin{itemize}\setlength{\itemsep}{2pt}
  \item \textbf{Bid-shape widening} in critical hours --- dispatchables write richer ladders where within-hour MCP variation is priceable. (Spec~C DiD.)
  \item \textbf{Within-hour wedge SD} jumps under ID15 and does not collapse at DA15 --- consistent with limited per-product arbitrage. (Spec~A.)
  \item \textbf{IDA sell-share} rises sharply at ID15 for nuclear, wind, pump-storage --- production-unit arbitrageurs reposition. (Spec~A.)
  \item \textbf{System cost} shrinks within the pre-blackout zone --- auction-stage granularity absorbs variation REE would otherwise clean up ex post. (Descriptive.)
\end{itemize}

\medskip
\textbf{What the model predicts and the data are consistent with but cannot identify:} a level price drop at the spot reform (M2), a Cournot scale-up of cleared MWh at the forward reform (mech.\ iv), in-band offered-energy migration. Point estimates have the right sign; CIs are wide once renewables vary year-by-year.

\medskip
\textbf{Reform compresses, does not amplify, market power.} Wedge mean stays at the small positive Ito-Reguant forward premium; the within-quarter strategic engagement is more competition, not more rent. Post-blackout cost rise is a separate \emph{reforzada} response, CCGT-only.

\medskip
\textbf{Microstructure $\Rightarrow$ macro.} The same finer clock that gives bidders within-hour discrimination also lowers system-level balancing cost.
\end{frame}
