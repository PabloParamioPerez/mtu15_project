\section{Introduction}

\begin{frame}{Objective and motivation}
  \begin{itemize}
    \item The purpose of my thesis is to empirically assess the impact of Spain's 2024--2025 \textbf{market time unit (MTU15) reform sequence in the electricity wholesale markets}, focusing on firm strategic behaviour and market outcomes and building a theoretical framework to better capture the mechanism.
    \item Due to the high institutional and data complexity of the reform, I have focused on building a robust data pipeline, as well as understanding the market structure and reform context in depth, before attempting to test any specific hypothesis.
    \item Moreover, we have to be very careful with how we read the data, since the reform sequence is not cleanly separated from other confounding events (gas-price normalisation, blackout and post-blackout regime, etc.), so \textbf{institutional awareness} is key here.
  \end{itemize}
\end{frame}

\section{Reference: identification design}

\begin{frame}{Slide 12 --- Identification: within-day critical-vs-flat hours DiD}
  \footnotesize
  \textbf{The core design idea.} The MTU15 reform doesn't change market rules uniformly across hours of the day --- it changes them \emph{differentially}. Granularity reform (60-min $\to$ 15-min) only bites economically in hours where the within-hour residual-demand profile is steep enough that 15-min slicing creates strategic optionality. In flat hours, 15-min and 60-min are mechanically equivalent.\\[4pt]
  \textbf{Within-day treatment-control structure:}
  \begin{itemize}
    \item \textbf{Treated unit:} critical hours $h \in \{7, 8, 16, 17, 18\}$. Top 5 by within-hour residual-demand variance ($\sigma^2_{\text{within}}$). Steep ramp hours where granularity should bite.
    \item \textbf{Control unit:} flat hours $h \in \{3, 4, 5\}$. Pre-dawn hours, bottom of $\sigma^2_{\text{within}}$ ranking. Genuinely flat within-hour profile, no strategic optionality from granularity.
    \item \textbf{Treatment timing:} reform regime indicator (DA15/ID15 vs pre-IDA).
    \item \textbf{Estimand:} the causal effect of the reform on the granularity mechanism, identified by how the critical-flat differential changes from pre-IDA to DA15/ID15.
  \end{itemize}
  \textbf{Specification:}
  \[ q_{ift} = \alpha + \beta_1 \,\text{crit}_h + \beta_2 \,\text{post}_t + \beta_3 \,(\text{crit}_h \times \text{post}_t) + \gamma_f + \delta_d + \varepsilon_{ift} \]
  Big-4 firm $f$, day $t$ (Day-of-Week FE $\delta_d$), hour $h$. Cluster-robust SE by date.
\end{frame}

\begin{frame}{Slide 13 --- Five identification advantages over standard reform DiD}
  \footnotesize
  \begin{enumerate}
    \item \textbf{Within-day FE absorbs all daily-level shocks.} Critical and flat hours of the same day share the same Spanish wind regime, the same temperature, the same hydro inflow, the same fuel prices, the same regulatory announcements. Anything operating at the daily level differences out by construction.
    \item \textbf{Calendar mix controls itself.} Critical and flat hours come from the same dates. Identical calendar mix for treated and control by construction.
    \item \textbf{Weather differences out at the hour-pair level.} Same-day hours share most weather conditions.
    \item \textbf{Built-in parallel-trends check.} The pre-IDA crit-flat differential is the placebo test: if pre-existing crit-flat structural differences are large and unstable, the design is in trouble.
    \item \textbf{Treatment assignment is mechanism-targeted.} The contrast is ``hours where granularity should matter'' vs ``hours where it shouldn't'' --- not ``post-reform vs pre-reform''. Identifies the granularity-mechanism causal effect specifically.
  \end{enumerate}
  \textbf{What within-day FE does NOT solve:} within-hour-of-day trends across calendar time. Spanish solar grew $\sim$6$\times$ since 2018; renewable share at h\{7,8,16,17,18\} vs h\{3,4,5\} has a calendar-time trend. Conditional parallel trends would require absorbing this. Tonight's test: it does \emph{not} hold even with wind+solar+demand $\times$ hour controls.
\end{frame}

\begin{frame}{Slide 14 --- Headline empirical result: q\_2 V-shape}
  \footnotesize
  \textbf{Outcome variable:} $q_2$ = signed Big-4 IDA repositioning per firm-day-hour, MWh. From PIBCIE, summing IDA-cleared assigned\_power\_mw $\times$ mtu\_minutes/60 across periods within each (date, firm, hour). Positive $q_2$ = firm sells more in IDA above DA commitment; negative = firm buys back.\\[4pt]
  \textbf{Spec:} $q_2 \sim \text{crit} + \text{post} + \text{crit}\times\text{post} + \text{firm FE} + \text{DOW FE}$, cluster by date. Treated $h\in\{7,8,16,17,18\}$, control $h\in\{3,4,5\}$. Pre-IDA = before 2024-06-14. Post = DA15/ID15 (Oct 2025+).\\[4pt]
  \textbf{Result with clean flat-hour controls:}
  \begin{center}
  \begin{tabular}{lrrr}
    \toprule
    Coefficient                         & Estimate     & SE      & p-value \\
    \midrule
    crit\_pre (pre-IDA crit-flat gap)   & $+41.21$     & $3.36$  & $1.6\!\times\!10^{-34}$ \\
    DiD $\delta$ (crit$\times$post)     & $+58.67$     & $12.18$ & $1.5\!\times\!10^{-6}$  \\
    \textbf{crit\_post} (DA15/ID15 crit-flat gap) & \textbf{$+99.80$} & --- & --- \\
    \bottomrule
  \end{tabular}
  \end{center}
  \textbf{Interpretation.} At DA15/ID15, Big-4 firms reposit \textbf{+100 MWh more} per firm-hour in IDA in critical hours than in flat (pre-dawn) hours. The reform-attributable component (DiD $\delta$) is +59 MWh.\\[2pt]
  Sample: 78,177 firm-day-hours. Two firms (GN, IB) carry most of the result; see slide 22.
\end{frame}

\begin{frame}{Slide 15 --- Same-calendar-month robustness}
  \footnotesize
  \textbf{The concern this addresses.} Pre-IDA spans 2018--2024 (all calendar months). DA15/ID15 spans Oct 2025+ (only Oct--May months). Calendar mix differs $\Rightarrow$ Spanish seasonality could bias the comparison.\\[4pt]
  \textbf{The fix.} Restrict pre-IDA observations to the same calendar months as DA15/ID15 (Oct--May only). Re-estimate the same DiD spec.\\[4pt]
  \textbf{Outcome:} $q_2$ as above (Big-4 IDA repositioning per firm-hour, MWh). Note: this run uses the original definition with all-non-critical-hours as control (the +83 baseline rather than +100 with clean controls).
  \begin{center}
  \begin{tabular}{lrrrrr}
    \toprule
    Pre-IDA window     & N\_pre   & DiD $\delta$  & SE   & p-value          & ratio vs full \\
    \midrule
    Full (default)     & 224,061  & $+79.36$      & 6.71 & $3\!\times\!10^{-32}$ & 100\% \\
    Same-cal-month     & ---      & $+76.5$       & 6.77 & $1\!\times\!10^{-29}$ & \textbf{96\%} \\
    \bottomrule
  \end{tabular}
  \end{center}
  \textbf{Verdict.} The headline coefficient survives same-calendar-month restriction at 96\% magnitude. Calendar mix is not driving the result.\\[2pt]
  CLAUDE.md verdict criterion (``$>$50\% magnitude with same sign'') easily met.
\end{frame}

  \textbf{Test.} Re-estimate the DiD with progressively narrower pre-IDA windows. With clean flat controls $h\{3,4,5\}$:

\begin{frame}{Slide 17 --- Per-hour DiD reveals two-cluster mechanism}
  \footnotesize
  \textbf{Per-hour DiD with control = $h\{3,4,5\}$, outcome = $q_2$.} For each hour $h$, treat that single hour vs the flat-hour control set. Reveals which hours actually carry the reform signal:
  \begin{center}
  \footnotesize
  \begin{tabular}{lrrrl}
    \toprule
    Cluster                    & hours              & DiD $\delta$ range & p-values             & sign \\
    \midrule
    Morning ramp + peak        & h7, h8             & $+64$ to $+68$    & $<10^{-6}$           & + \\
    Evening peak               & h17, h18, h19      & $+68$ to $+101$   & $<10^{-7}$           & + \\
    \textbf{Midday plateau}    & h10--h15           & $-48$ to $-131$   & $<10^{-7}$           & \textbf{$-$} \\
    Late-night transition      & h0--h2             & $+25$ to $+46$    & $<10^{-4}$           & + (weak) \\
    Other                      & h6, h9, h16, h20--h22 & near zero       & ns                   & --- \\
    h23 (late edge)            & h23                & $-70$             & $5\!\times\!10^{-6}$ & $-$ \\
    \bottomrule
  \end{tabular}
  \end{center}
  \textbf{Two-cluster mechanism:}
  \begin{itemize}
    \item \textbf{Ramp + evening-peak hours: positive DiD $\delta$.} Big-4 reposit MORE in IDA under DA15/ID15. Consistent with Ito--Reguant: residual demand rising sharply within the hour creates strategic withholding amplified by 15-min granularity.
    \item \textbf{Midday plateau (h10--h15): negative DiD $\delta$, large magnitude.} Big-4 reposit LESS (or buy back) in IDA. Solar abundance crowds out conventional commitment; 15-min granularity allows precise quarter-by-quarter reduction in commitment.
  \end{itemize}
  \textbf{Implication.} The mechanism is bidirectional --- granularity amplifies critical-hour repositioning, in both signs depending on whether residual demand is rising (scarcity) or falling (abundance) within the hour.
\end{frame}

\begin{frame}{Slide 18 --- Cumulative critical-set sensitivity}
  \footnotesize
  \textbf{Why this matters.} The canonical 5-hour critical set comes from a top-5 cut of $\sigma^2_{\text{within}}$. Why stop at 5? The $\sigma^2_{\text{within}}$ ranking after rank 5 is h\{12, 6, 9, 11, 13, 14, 10\}. Adding any of those should clarify if 5 is the right cut.\\[4pt]
  \textbf{Cumulative DiD with control = $h\{3,4,5\}$, outcome = $q_2$:}
  \begin{center}
  \footnotesize
  \begin{tabular}{lrrrr}
    \toprule
    Top $K$ critical hours & added                            & DiD $\delta$ (SE)   & crit\_post & p-value \\
    \midrule
    K=1: \{17\}            & ---                              & $+67.7\,(16.7)$     & $+113.5$    & $5\!\times\!10^{-5}$ \\
    K=4: \{7,16,17,18\}    & ramp + evening                   & $+57.4\,(12.7)$     & $+100.5$    & $6\!\times\!10^{-6}$ \\
    \textbf{K=5: canonical \{7,8,16,17,18\}}    & + h8 (morning) & $+58.6\,(12.2)$     & \textbf{$+99.8$} & $2\!\times\!10^{-6}$ \\
    K=6: + h12             & adds midday                      & $+30.5\,(11.6)$     & $+71.1$     & $9\!\times\!10^{-3}$ \\
    K=11: + h\{6,9,11--14\}& full midday + ramp neighbours    & $-11.4\,(10.2)$     & $+26.7$     & ns \\
    K=12: + h10            & also midday                      & $-14.5\,(10.1)$     & $+23.5$     & ns \\
    \bottomrule
  \end{tabular}
  \end{center}
  \textbf{Verdict.} The +100 result is robust to dropping any single canonical hour, and to cutting at K=4 (without h8) or K=1 (just h17). \textbf{Adding h12 (rank 6) collapses the coefficient by 30\%}, because h12 is in the opposite-sign midday cluster. The $\sigma^2_{\text{within}}$ ranking is a necessary but not sufficient hour selector --- it doesn't distinguish ``hours where granularity creates rising-demand scarcity'' from ``hours where granularity creates falling-demand abundance''. The canonical 5-hour set happens to align cleanly with the rising-demand cluster.
\end{frame}
