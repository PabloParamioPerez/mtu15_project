% ---- 3. Motivation ----
\section{Introduction}

\begin{frame}{Motivation}
  \begin{itemize}
    \item Empirically assess the impact of Spain's 2024--2025 \textbf{market time unit (MTU15) reform sequence} on \emph{firm strategic behaviour} and \emph{market outcomes}.
    \vspace{4pt}
    \item Why interesting: the reform changes the time-unit granularity of the wholesale market in three stages, with \emph{transitional asymmetric-clock windows} where settlement and trading run on different clocks. A useful setting to study how granularity shapes firm conduct.
    \vspace{4pt}
    \item Why it is hard: the reform sequence overlaps with confounders --- gas-price normalisation (Dec 2023), the 2025-04-28 blackout and post-blackout \emph{operación reforzada} (reinforced operation), the elimination of CNMC Rule 28.8 for bilateral contracts (Mar 2025). \textbf{Institutional awareness is essential.}
    \vspace{4pt}
    \item This presentation: identification strategy, headline result on \textbf{dominant firms} (Iberdrola, Endesa, Naturgy, EDP) strategic intraday repositioning, and an honest account of what survives robustness and what does not.
  \end{itemize}
\end{frame}

% ----- 6. Economic mechanisms ----
\section{Economic mechanisms}

\begin{frame}{Theoretical anchors and our mechanism}
  \footnotesize
  \begin{block}{Allaz \& Vila (1993, JET) --- forward markets dispel rents}
    Adding more forward markets prior to the spot stage commits Cournot firms to quantities and erodes withholding incentives. \emph{Caveat:} A--V adds \textbf{more markets for the same good} (e.g., a second forward stage). Our reform \textbf{divides one good into four sub-goods} (an hour into 4 quarters). \textbf{Related intuition; not the same mechanism.}
  \end{block}
  \begin{block}{Ito \& Reguant (2016, AER) --- sequential markets and arbitrage}
    Firms with market power reposition across sequential markets (under-commit in DA, sell more in IDA). \emph{Caveat:} I--R is about \textbf{sequencing across markets}; we are about \textbf{granularity within a market}. \textbf{Related, not equivalent.}
  \end{block}
  \begin{block}{Chang (2026, WP) --- settlement vs dispatch period mismatch}
    When settlement period differs from dispatch period, averaging effects change bidding incentives. \emph{Does not apply to DA15/ID15}: settlement and dispatch are both 15-min --- no averaging. Only the ISP15-with-DA60 transition window had averaging effects, and only on \emph{quantities} (imbalance settlement averaged within the hour).
  \end{block}
  \begin{block}{Our preliminary mechanism --- within-market granularity (what we test)}
    Splitting an hour into 4 quarters that clear independently lets firms bid different prices and quantities across quarters of the same hour. \textbf{Especially valuable when residual demand varies steeply within the hour} (morning ramps, evening peaks). In flat hours where within-hour demand is constant, granularity has no economic content. \textbf{To be formalized!}
  \end{block}
\end{frame}


% ----- 7. What we want to test, 8. Data, 9. Bid-shape ----
\section{What we want to test, data, and bid-shape evidence}

\begin{frame}{What we want to test, and how}
  \footnotesize
  \textbf{Our mechanism predicts.} Dominant firms exploit within-hour granularity in \emph{critical} hours (steep within-hour ramps in residual demand) but not in \emph{flat} hours (constant within-hour profile). The reform should differentially affect critical vs flat hours.\\[4pt]
  \textbf{Outcomes:}
  \begin{itemize}
    \item $q_2$ = dominant firms' \emph{voluntary} IDA repositioning per firm-day-hour (signed PIBCIE), MWh. \textbf{Headline outcome.}
    \item $q_1$ = dominant firms' day-ahead auction-cleared sell, MWh per firm-hour.
    \item Bid-shape: tranche count, ladder slope, qty-weighted price (post-MTU15-DA only).
  \end{itemize}
  \textbf{How we define critical and flat hours.} For each hour-of-day $h$, compute $\sigma^2_{\text{within}}(h)$ = \textbf{the variance of conventional production (load $-$ wind $-$ solar) across the four 15-min quarters of hour $h$}, averaged over days. High $\sigma^2_{\text{within}}$ means dispatchable units need to ramp a lot \emph{within} the hour --- exactly where within-hour granularity has economic content.
  \begin{itemize}
    \item \textbf{Treated} = critical hours $h\in\{7,8,16,17,18\}$. Top-5 by $\sigma^2_{\text{within}}$ (morning ramp + evening peak).
    \item \textbf{Control} = flat hours $h\in\{3,4,5\}$. Bottom of $\sigma^2_{\text{within}}$ (pre-dawn).
  \end{itemize}
  \textbf{Why flat hours are a valid counterfactual.} If the granularity reform has \emph{zero economic content} in flat hours (because there is nothing within the hour to slice strategically), then comparing critical-hour behaviour to flat-hour behaviour identifies the mechanism. \emph{This is a hypothesis I need to defend with a small theoretical model showing that, \textbf{in equilibrium}, the reform has no effect for flat hours --- ongoing.}
\end{frame}

\begin{frame}{Data}
  \footnotesize
  \textbf{Coverage:} 8 years (2018-01 $\to$ 2026-04), at native 60-min/15-min granularity throughout.\\[4pt]
  \textbf{Three sources:}
  \begin{itemize}
    \item \textbf{OMIE (Iberian market operator)}: prices and \emph{bid-level} data for DA, IDA, continuous; per-firm and per-unit programmes. $\sim$5.7 billion rows ($\sim$413 GB raw, $\sim$198 GB processed).
    \item \textbf{ENTSO-E Transparency Platform}: imbalance volumes and prices, balancing settlement, generation by type/unit, load and renewable forecasts. All system-level fundamentals come from here.
    \item \textbf{ESIOS (REE)}: Spanish per-segment settlement, real-time prices, balancing offers. \textbf{API token requested 3 weeks ago; still pending due to technical issues on the operator's side.} Reserves attribution at firm level partially blocked.
  \end{itemize}
  \textbf{Engineering principles:}
  \begin{itemize}
    \item Pipelines are \textbf{idempotent}: re-running any step produces the same output and includes new data --- supports replication.
    \item 25+ data families, each with its own parser and tests.
    \item Empirical findings tracked formally (alive / wounded / dead) with a documented audit trail.
  \end{itemize}
  \textbf{Honest caveat.} Most of the empirical findings in the audit log are descriptive rather than carrying clear economic interest --- e.g., the structural break in Q1 2024 nuclear bilateral share. The presentation focuses on the small subset that has both economic content and clean identification.
\end{frame}

\begin{frame}{Bid-shape: dominant firms exploit granularity (justifies the design)}
  \footnotesize
  Direct evidence at the bid level that operators \emph{actually use} within-hour granularity post-MTU15-DA --- this justifies using flat hours as a valid counterfactual.\\[4pt]
  \textbf{What we observe.} For dominant-firm CCGT in critical hours (post-MTU15-DA):
  \begin{itemize}
    \item More price tranches per quarter ($+8\%$ vs flat hours).
    \item Steeper ladder slope ($+13\%$).
    \item \emph{Lower} qty-weighted average price.
  \end{itemize}
  \textbf{Interpretation.} Operators bid a richer ladder in critical hours: low-price inframarginal tranches at the bottom (push the qty-weighted average down, lock clearing) plus a steeper high-price tail (capture scarcity rent). They are unambiguously exploiting within-hour granularity.\\[4pt]
  \textbf{Tech selectivity.} Concentrated in CCGT and (mildly) hydro. Wind, solar, nuclear show no engagement. Crucially: many plants \textbf{mechanically repeat their previous hourly bid identically across all 4 quarters} --- not exploiting the granularity at all.\\[4pt]
  \textbf{Why this matters for the design.} Bids confirm operators \emph{can} and \emph{do} use within-hour granularity in the right hours. Where granularity is exploited (critical), we should see effects on quantities; where not (flat), we should not.
\end{frame}


% ----- 10. DiD spec, 11. Headline result ----
\section{Identification and main result}

\begin{frame}{The DiD specification}
  \footnotesize
  \textbf{Panel.} The unit of observation is (firm $f$, day $d$, hour $h$). We aggregate across the 4 quarters within an hour. So the outcome is $q_{fdh}$.\\[4pt]
  \textbf{Specification (very preliminary):}
  \[ q_{fdh} = \alpha + \beta_1 \,\text{crit}_h + \beta_2 \,\text{post}_d + \beta_3 \,(\text{crit}_h \times \text{post}_d) + \gamma_f + \delta_{\text{DOW}(d)} + \varepsilon_{fdh} \]
  $\beta_3$ is the DiD coefficient. $\gamma_f$ are firm fixed effects (Iberdrola, Endesa, Naturgy, EDP); $\delta_{\text{DOW}}$ are day-of-week fixed effects. Cluster-robust SE by date $d$.\\[4pt]
  \textbf{Window in each regression:}
  \begin{itemize}
    \item \textbf{Pre} = pre-DA15 reform baseline: 2018-01-01 $\to$ 2024-06-13.
    \item \textbf{Post} = DA15/ID15: 2025-10-01 $\to$ present.
    \item Intermediate regimes (3-sess, ISP15-win, DA60/ID15) handled separately when needed.
  \end{itemize}
  \textbf{Why the DiD removes time-varying confounders:}
  \begin{itemize}
    \item Critical and flat hours come from the \emph{same dates} $\Rightarrow$ daily shocks (weather, demand level, fuel prices, regulatory announcements, macro conditions) absorb mechanically in the within-day comparison.
    \item Pre-DA15-reform crit-flat differential serves as the parallel-trends sanity check.
    \item The contrast we identify is \emph{mechanism-targeted}: hours where granularity should bite vs hours where it should not.
  \end{itemize}
  \textbf{Heterogeneity.} Most of the response carried by Naturgy (CCGT $+$ Hydro). Iberdrola contributes through nuclear (caveat: allocation artefact). Endesa and EDP near-zero. Different technologies respond very differently --- consistent with the bid-shape evidence on the previous slide.
\end{frame}

\begin{frame}{Headline result: dominant firms reposition more in critical hours}
  \footnotesize
  \textbf{Same panel and spec as previous slide:} outcome $q_{fdh}$ (signed IDA repositioning, MWh per firm-day-hour); $\text{crit}_h = 1$ for $h\{7,8,16,17,18\}$, $0$ for $h\{3,4,5\}$; $\text{post}_d = 1$ from 2025-10, $0$ for the pre-DA15-reform baseline.\\[3pt]
  \begin{center}
  \begin{tabular}{lrrr}
    \toprule
                                                                 & Estimate           & SE      & p-value \\
    \midrule
    $\beta_1$ \,(crit$_h$, pre-DA15-reform crit$-$flat gap)      & $+41.21$           & $3.35$  & $1.1\!\times\!10^{-34}$ \\
    $\beta_2$ \,(post$_d$, flat-hour level shift pre$\to$post)   & $-78.53$           & $10.40$ & $4.3\!\times\!10^{-14}$ \\
    $\beta_3$ \,(crit$_h \times$ post$_d$, DiD)                  & $\mathbf{+58.60}$  & $12.18$ & $1.5\!\times\!10^{-6}$  \\
    \bottomrule
  \end{tabular}\\[2pt]
  \footnotesize $N = 78{,}232$ firm-day-hours; cluster-robust SE by date.
  \end{center}
  \textbf{Type I / type II tradeoff.} With $N \approx 80\text{k}$, statistical power is very high (low type II error), so the conventional $p < 0.05$ threshold becomes too lenient and inflates type I error. Worth thinking carefully about how to set significance thresholds in this regime.\\[3pt]
  \textbf{Why these hours.} The top-5 critical and bottom-3 flat are the hours where the robustness battery (per-hour DiD, cumulative critical-set sensitivity) confirms the effect most reliably --- we use them as the cleanest contrast.\\[3pt]
  \textbf{Interpretation.} The outcome $q_2$ (signed IDA repositioning above DA commitment) is the empirical signature \textbf{Ito \& Reguant (2016) use to detect strategic intra-day arbitrage}: positive $q_2$ means firms under-commit in DA and resell in IDA. We adopt their measure but ask a \emph{different} question: \textbf{does this strategic IDA repositioning grow more in critical hours after the reform splits each hour into 4 sub-periods?} The DiD coefficient $\beta_3 = +59$ says yes. Mechanism: \textbf{the reform lets firms bid different prices/quantities for each quarter of an hour. This pays off where the within-hour production profile is steep --- so firms exploit the ramps. Flat hours have no within-hour variation, so no effect.}
\end{frame}


% ----- 12. Honest assessment (robustness moved to appendix) ----
\section{Honest assessment}

\begin{frame}{What is robust, what is not}
  \footnotesize
  \textbf{Robust:}
  \begin{itemize}
    \item The within-DA15/ID15 fact: \textbf{$+100$ MWh more IDA repositioning in critical vs flat hours per firm-hour.}
    \item \textbf{Fringe placebo:} same DiD on top non-dominant firms (EV, DET, EGL, EHN, REP) gives $\beta_3 = -24.27$ (SE 8.07, $p = 2.7\!\times\!10^{-3}$) --- opposite-signed, consistent with the strategic-mechanism interpretation (mechanism requires market power).
    \item Bid-shape rich-ladder pattern in CCGT post-MTU15-DA.
    \item Two-cluster mechanism: ramp $+$ peak positive; midday negative.
  \end{itemize}
  \textbf{Not robust:}
  \begin{itemize}
    \item The reform-attributable component (DiD $\delta$) is a range, not a point estimate. Pre-trend instability is real.
    \item Conditional parallel trends with renewable $+$ demand controls does not hold.
    \item Heterogeneity: Naturgy carries most of the result; cross-firm response is uneven.
  \end{itemize}
  \textbf{The theoretical-model gap.} The design needs a stylised IO model showing that, in equilibrium, the within-market granularity reform is a no-op for flat hours --- this is what justifies using them as control. The model is anchored in \emph{our} mechanism (bidding differently across the 4 quarters of an hour, exploiting within-hour ramps), not directly in Allaz--Vila or Ito--Reguant. \textbf{Out of scope}: a fully estimated structural auction model.
\end{frame}

\begin{frame}{What is left}
  \footnotesize
  \begin{itemize}
    \item \textbf{Stylised theoretical model} that justifies flat hours as control --- the foundation of the empirical design.
    \item \textbf{Better identification strategy.} The clean flat-hour design came together yesterday; there is more thinking to do.
    \item Think about the demand side more --- the current mechanism and design are supply-side focused.
    \item Include the continuous market in the analysis.
    \item Additional robustness checks once the mechanism interpretation is locked.
    \item Per-segment settlement once the ESIOS API token clears, but not necessary a bottleneck for the main result.
  \end{itemize}
\end{frame}

\section{Appendix}

\begin{frame}{Appendix --- robustness checks (in words)}
  \footnotesize
  \begin{itemize}
    \item \textbf{Same-calendar-month restriction.} Restricting the pre-DA15-reform baseline to the same months as the post window (Oct--May) leaves the result at $96\%$ of the magnitude. Calendar mix not the driver.
    \item \textbf{Baseline-window sensitivity.} Across 5 narrower pre-DA15-reform windows, the within-DA15/ID15 fact ($+100$) is constant; reform-attribution component ranges $+27$ to $+133$. \textbf{To think about:} pre-trend in the crit-flat gap is not stable.
    \item \textbf{Per-hour decomposition.} Treating each hour individually shows two clusters with opposite signs (ramps + peak positive; midday negative). Justifies the 5-hour critical set.
    \item \textbf{Conditional parallel trends.} Wind $+$ solar $+$ demand $\times$ hour controls absorb only $\sim 13\%$ of the pre-trend. \emph{Does NOT hold.} \textbf{To think about:} an unobserved time trend (CCGT retirement, regulatory anticipation) is at play.
    \item \textbf{Bid-shape and extensive-margin.} Unit-level confirmation: richer bid ladders in critical hours (CCGT only); 13 CCGT units inactive in the asymmetric-clock window, return under \emph{operación reforzada}.
  \end{itemize}
\end{frame}

\begin{frame}{Appendix --- fringe placebo}
  \footnotesize
  \textbf{Same DiD spec, run on fringe firms} (top non-dominant firms in PIBCIE: EV, DET, EGL, EHN, REP). If our mechanism (within-market granularity exploited by firms with pivotal market power) is correct, fringe firms --- by definition without pivotal market power --- should NOT show the same response.\\[4pt]
  \begin{center}
  \begin{tabular}{lrrr}
    \toprule
                                                  & Estimate          & SE      & p-value \\
    \midrule
    \textbf{Dominant firms} (IB, GE, GN, EDP)     & $\mathbf{+58.60}$ & $12.18$ & $1.5\!\times\!10^{-6}$ \\
    Fringe (EV, DET, EGL, EHN, REP)               & $-24.27$          & $8.07$  & $2.7\!\times\!10^{-3}$ \\
    \bottomrule
  \end{tabular}\\[2pt]
  \footnotesize $\beta_3$ (crit$_h \times$ post$_d$) only; same outcome ($q_2$, MWh/firm-hour) and same window. Dominant N $= 78{,}232$; fringe N $= 46{,}327$.
  \end{center}
  \textbf{Reading.} Dominant firms reposition $+59$ MWh more in critical hours after the reform. Fringe firms go the \emph{opposite direction} ($-24$ MWh, $p < 0.01$). The mechanism asks for market power, and the placebo confirms it: where the strategic optionality of within-hour granularity is not exercisable (no pivotal market power), the response is null or opposite-signed --- consistent with fringe firms being crowded out of strategic positions by the dominant firms' more aggressive critical-hour bidding.
\end{frame}
